\begin{exercise}[Technical]{Arithmetic expansion}
    \vspace{-5mm}
    \begin{enumerate}
        \item Using arithmetic expansion only in your terminal, find the largest integer which can be stored in a Bash variable.
        \item How many bits are used in your opinion to store an integer value?
        \item How would you explain that \bash{echo $(( 2**64 ))} prints \texttt{0}? Try to change \bash{64} with larger integers.
        \item Read section 6.5 of the \URL[Navy]{https://www.gnu.org/software/bash/manual/}{Bash manual v5.0} carefully.\\
              Is there any difference between \mbox{\bash{|+$(( 36#a ))+|}} and \mbox{\bash{|+$(( 36#A ))+|}}?
              And between \mbox{\bash{|+$(( 37#a ))+|}} and \mbox{\bash{|+$(( 37#A ))+|}}?
        \item Can you explain the output of the following command?
              \begin{lstlisting}[style=MyBash]
                  printf "%d %x %o\n" |+$(( 64#_ )) $(( 64#_ )) $(( 64#_ ))+|
              \end{lstlisting}
    \end{enumerate}
    \vspace{-0mm}
\end{exercise}
